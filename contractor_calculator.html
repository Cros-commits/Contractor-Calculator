    <script>

        function updateTable() {

            // Get the day rate from the input field

            let dayRate = parseFloat(document.getElementById('dayRate').value);

            

            // Calculate deductions

            let marketAdjustment = dayRate * 0.05;

            let onCosts = dayRate * (0.07 + 0.0045 + 0.0075 + 0.013); // Sum of payroll tax, insurance, severance, leave loading

            let payrollTax = dayRate * 0.07;

            let insurance = dayRate * 0.0045;

            let severance = dayRate * 0.0075;

            let leaveLoading = dayRate * 0.013;


            // Calculate the true day rate

            let trueDayRate = dayRate - marketAdjustment - onCosts;


            // Calculate total value earned (assuming 210 worked days)

            let totalValue = trueDayRate * 210;


            // Update the table with the calculated values

            document.getElementById('marketAdjustment').textContent = `($${marketAdjustment.toFixed(2)})`;

            document.getElementById('onCosts').textContent = `($${onCosts.toFixed(2)})`;

            document.getElementById('payrollTax').textContent = `$${payrollTax.toFixed(2)}`;

            document.getElementById('insurance').textContent = `$${insurance.toFixed(2)}`;

            document.getElementById('severance').textContent = `$${severance.toFixed(2)}`;

            document.getElementById('leaveLoading').textContent = `$${leaveLoading.toFixed(2)}`;

            document.getElementById('trueDayRate').textContent = `$${trueDayRate.toFixed(2)}`;

            document.getElementById('totalValue').textContent = `$${totalValue.toFixed(2)}`;

        }


        // Initialize the table with default values

        updateTable();

    </script>
